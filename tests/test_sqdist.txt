>>> from lurrn.sparsmat import *
>>> a = VecD2()
>>> a.add_count(0,0)
>>> a.add_count(0,0)
>>> a.add_count(0,1)
>>> a.add_count(0,2)
>>> a.add_count(1,0)
>>> a.add_count(1,1)
>>> a.add_count(1,3)
>>> b = a.to_csr()
>>> v1 = b[0]
>>> v2 = b[1]
>>> v1  
SparseVectorD([(0,2.0),(1,1.0),(2,1.0)])
>>> v2  
SparseVectorD([(0,1.0),(1,1.0),(3,1.0)])
>>> v1.sqdist(v2)  
3.0
>>> v2.sqdist(v1)  
3.0
